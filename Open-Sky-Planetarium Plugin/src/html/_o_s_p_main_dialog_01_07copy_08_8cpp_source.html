427  else if(com[0].compare("turn")==0){
428  if(com[1].compare("on")==0){
429  emit comTURN(true);
430  }
431  else{
432  emit comTURN(false);
433  }
434  }
435  else if(com[0].compare("play")==0){
436  emit comPLAY(com[1]);
437  }
438  }
439  }
440 }

443 /*
444 compileScript():
445  This function is used to compile script
446  this function maps the user script commands to the C++ functions
447 */
448 void OSPMainDialog :: compileScript(){
449  QRegExp scriptExp("^([goto | wait | turn | play]{4})\\s*([\u0370-\u03FF0-9a-zA-Z\\.\\s]*)\\s*;$");
450  bool compileStatus = true;
451  int line_count=0;
452  commandsList.clear();
453  for(QString w : ui->scriptEdit->toPlainText().split("\n")){
454  line_count++;
455  if(!scriptExp.exactMatch(w) && w.compare("")!=0){
456  compileStatus=false;
457  showMessage(QString("Syntax Error at line %1").arg(line_count));
458  }
459  else{
460  QStringList com_par=scriptExp.capturedTexts();
461  if(com_par[1].compare("goto")==0){
462  double ra=0.0,dec=0.0;
463  QString sra,sdec;
464  //Todo:add checks before going down
465  StelObjectP star = GETSTELMODULE(StelObjectMgr)->searchByNameI18n(com_par[2].trimmed());
466  if(!star.isNull()){
467  StelUtils::rectToSphe(&ra,&dec,star->getEquinoxEquatorialPos(StelApp::getInstance().getCore()));
468  sra.setNum(ra);
469  sdec.setNum(dec);
470  commandsList << QString("goto_"+sra+"_"+sdec);
471  }
472  else{
473  showMessage("Could not find the star with the given name");
474  }
475  }
476  else if(com_par[1].compare("turn")==0){
477  if(com_par[2].trimmed().compare("on")!=0 && com_par[2].trimmed().compare("off")!=0){
478  compileStatus=false;
479  showMessage(QString("The parameter for lasr must be either 'on' or 'off' at line %1").arg(line_count));
480  break;
481  }
482  else{
483  commandsList << QString("turn-"+com_par[2].trimmed());
484  }
485  }
486  else if(com_par[1].compare("play")==0){
487  QFile f(QString(audioDirectoryPath+"/"+com_par[2].trimmed()));
488  if(!f.exists()){
489  compileStatus=false;
490  showMessage(QString("File not found in [{Stellarium User Directory}/modules/OpenSkyPlanetarium/audio/] at line %1").arg(line_count));
491  break;
492  }
493  else{
494  commandsList << QString("play-"+com_par[2].trimmed());
495  }
496  }
497  else if(com_par[1].compare("wait")==0){
498  QRegExp a("([1-5]?[0-9])\\s*m\\s*([1-5]?[0-9])\\s*s");
499  if(!a.exactMatch(com_par[2].trimmed())){
500  showMessage(QString("(X)m(Y)s format expected as argument for wait at line %1").arg(line_count));
501  compileStatus=false;
502  break;
503  }
504  else{
505  QStringList temp = a.capturedTexts();
506  commandsList << QString("wait-"+temp[1].trimmed()+"-"+temp[2].trimmed());
507  }
508  }
509  }
510  }
511  if (!compileStatus){
512  commandsList.clear();
513  }
514  else{
515  showMessage("Build Successful");
516  }
517  buildStatus=compileStatus;
518 }
