64 void SerialCom::run()
65 {
66  bool myPortNameChanged = false;
67 
68  mutex.lock();
69  QString myPortName;
70  if (myPortName != port) {
71  myPortName = port;
72  myPortNameChanged = true;
73  }
74 
75  int myWaitTimeout = waitTime;
76  QString myRequest = request;
77  mutex.unlock();
78  QSerialPort serial;
79 
80  while (!quit) {
81  if (myPortNameChanged) {
82  serial.close();
83  serial.setPortName(myPortName);
84 
85  if (!serial.open(QIODevice::ReadWrite)) {
86  emit error(tr("Can't open %1, error code %2")
87  .arg(port).arg(serial.error()));
88  return;
89  }
90  }
91  QByteArray requestData = myRequest.toLocal8Bit();
92  serial.write(requestData);
93  if (serial.waitForBytesWritten(waitTime)) {
94  if (serial.waitForReadyRead(myWaitTimeout)) {
95  QByteArray responseData = serial.readAll();
96  while (serial.waitForReadyRead(10))
97  responseData += serial.readAll();
98  QString response(responseData);
99  emit this->response(response);
100  } else {
101  emit timeout(tr("Wait Read Request Timed Out %1")
102  .arg(QTime::currentTime().toString()));
103  }
104  } else {
105  emit timeout(tr("Wait Write Request Timed Out %1")
106  .arg(QTime::currentTime().toString()));
107  }
108  mutex.lock();
109  cond.wait(&mutex);
110  if (myPortName != port) {
111  myPortName = port;
112  myPortNameChanged = true;
113  } else {
114  myPortNameChanged = false;
115  }
116  myWaitTimeout = waitTime;
117  myRequest = request;
118  mutex.unlock();
119  }
