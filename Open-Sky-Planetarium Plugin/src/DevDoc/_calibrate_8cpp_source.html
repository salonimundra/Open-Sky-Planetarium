56 /*
57  * Calculates the inverse of the m[3x3] matrix and returns it in the second parameter.
58  */
59 void Calibrate::_inv(double m[3][3], double res[3][3]){
60  double idet;
61 
62  //Inverse of the determinant
63  idet = 1/(
64  (m[0][0]*m[1][1]*m[2][2]) + (m[0][1]*m[1][2]*m[2][0]) + (m[0][2]*m[1][0]*m[2][1])
65  - (m[0][2]*m[1][1]*m[2][0]) - (m[0][1]*m[1][0]*m[2][2]) - (m[0][0]*m[1][2]*m[2][1])
66  );
67 
68  res[0][0] = ((m[1][1]*m[2][2]) - (m[2][1]*m[1][2]))*idet;
69  res[0][1] = ((m[2][1]*m[0][2]) - (m[0][1]*m[2][2]))*idet;
70  res[0][2] = ((m[0][1]*m[1][2]) - (m[1][1]*m[0][2]))*idet;
71 
72  res[1][0] = ((m[1][2]*m[2][0]) - (m[2][2]*m[1][0]))*idet;
73  res[1][1] = ((m[2][2]*m[0][0]) - (m[0][2]*m[2][0]))*idet;
74  res[1][2] = ((m[0][2]*m[1][0]) - (m[1][2]*m[0][0]))*idet;
75 
76  res[2][0] = ((m[1][0]*m[2][1]) - (m[2][0]*m[1][1]))*idet;
77  res[2][1] = ((m[2][0]*m[0][1]) - (m[0][0]*m[2][1]))*idet;
78  res[2][2] = ((m[0][0]*m[1][1]) - (m[1][0]*m[0][1]))*idet;
79 }

85 void Calibrate::_m_prod(double m1[3][3], double m2[3][3], double res[3][3]){
86  for(int i=0; i<3; i++)
87  for(int j=0; j<3; j++){
88  res[i][j] = 0.0;
